#!/usr/bin/python3

import unittest
import os
from append_to_file import append_to_file

class TestAppendToFile(unittest.TestCase):
    def setUp(self):
        self.filename = "test_append_file.txt"

    def tearDown(self):
        if os.path.exists(self.filename):
            os.remove(self.filename)

    def test_append_to_file(self):
        text1 = "This is the first part of the text."
        text2 = "This is the second part of the text."
        append_to_file(self.filename, text1)
        append_to_file(self.filename, text2)

        with open(self.filename, "r", encoding="utf-8") as file:
            content = file.read()

        self.assertEqual(content, text1 + text2)

if __name__ == '__main__':
    unittest.main()
